1 /*
2  * File: bitstream.h
3  * -----------------
4  * This file defines the ibitstream and obitstream classes which are basically
5  * same as the ordinary istream and ostream classes, but add the
6  * functionality to read and write one bit at a time.
7  *
8  * The idea is that you can substitute an ibitstream in place of an
9  * istream and use the same operations (get, fail, >>, etc.)
10  * along with added member functions of readBit, rewind, and size.
11  *
12  * Similarly, the obitstream can be used in place of ofstream, and has
13  * same operations (put, fail, <<, etc.) along with additional
14  * member functions writeBit and size.
15  *
16  * There are two subclasses of ibitstream: ifbitstream and istringbitstream,
17  * which are similar to the ifstream and istringstream classes. The
18  * obitstream class similarly has ofbitstream and ostringbitstream as
19  * subclasses.
20  *
21  * @author Keith Schwarz, Eric Roberts, Marty Stepp
22  * @version 2016/11/12
23  * - made toPrintable non-static and visible
24  */

50  * Class: ibitstream
51  * ---------------
52  * Defines a class for reading files with all the functionality of istream
53  * along with an added member function for reading a single bit and convenience
54  * functions for rewinding the stream back to the beginning and getting the stream
55  * size.
56  *
57  * You will probably not create instances of this class directly. Instead, you
58  * will create ifbitstreams or istringbitstreams to read from files or string buffers.
59  */
60 class ibitstream: public std::istream {
61 public:
62  /*
63  * Constructor: ibitstream
64  * Usage: ibitstream stream;
65  * -----------------------
66  * Initializes a new ibitstream that is not attached to any source. You are
67  * unlikely to use this function directly.
68  */
69  ibitstream();
70 
71  /*
72  * Member function: readBit
73  * Usage: bit = in.readBit();
74  * --------------------------
75  * Reads a single bit from the ibitstream and returns 0 or 1 depending on
76  * the bit value. If the stream is exhausted, EOF (-1) is returned.
77  * Raises an error if this ibitstream has not been properly opened.
78  */
79  int readBit();
80 
81  /*
82  * Member function: rewind
83  * Usage: in.rewind();
84  * -------------------
85  * Rewinds the ibitstream back to the beginning so that subsequent reads
86  * start again from the beginning. Raises an error if this ibitstream
87  * has not been properly opened.
88  */
89  void rewind();

91  /*
92  * Sets 'fake' mode, where it actually reads bytes when you say readBit.
93  */
94  void setFake(bool fake);
95 
96  /*
97  * Member function: size
98  * Usage: sz = in.size();
99  * ----------------------
100  * Returns the size in bytes of the data attached to this stream.
101  * Raises an error if this ibitstream has not been properly opened.
102  */
103  long size();
104
